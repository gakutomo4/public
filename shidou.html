<script>

    function divideSweets() {

        const totalSweets = parseInt(document.getElementById('totalSweets').value);

        const dividingKids = parseInt(document.getElementById('dividingKids').value);

        const resultDiv = document.getElementById('result');


        if (isNaN(totalSweets) || isNaN(dividingKids) || dividingKids <= 0) {

            resultDiv.innerHTML = '<p>正しい数を入力してね！</p>';

            return;

        }


        const quotient = Math.floor(totalSweets / dividingKids);

        const remainder = totalSweets % dividingKids;


        let resultHTML = `<p><strong>${totalSweets}個のお菓子を${dividingKids}人で分けると...</strong></p>`;

        

        resultHTML += `<div class="result-item"><p class="result-text">1人あたり</p>`;

        for (let i = 0; i < quotient; i++) {

            resultHTML += `<img src="https://emojicdn.elk.sh/🍰" alt="お菓子">`;

        }

        resultHTML += `<p class="result-text">もらえるね！</p></div>`;


        if (remainder > 0) {

            resultHTML += `<div class="result-item"><p class="result-text">そして、お菓子が</p>`;

            for (let i = 0; i < remainder; i++) {

                resultHTML += `<img src="https://emojicdn.elk.sh/🍩" alt="あまりのお菓子">`;

            }

            resultHTML += `<p class="result-text">あまりました。</p></div>`;

        } else {

            resultHTML += `<p>お菓子はあまりませんでした！みんなでぴったり分けられたね！</p>`;

        }


        resultDiv.innerHTML = resultHTML;

    }

</script>
